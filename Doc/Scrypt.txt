#  Описание формата скрипта Simple Game Engine 1.19
#
#  Общая форма записи команд:
#    [Команда] <Параметр 1> <Параметр N>
#
#  Пример:
#    SetParam S 'Пример параметра' - Установить парметр
#    Caption @S                    - Изменить заголовок окна на значение параметра S
#
#  Описание:
#    Любая строка начинающаяся с символа "#" считается заметкой.
#    Заметки не могут распологаться на одной строке с командами.
#    Используя Символ признака параметра "@", происходит подстановка его
#    значения в команду. Подстановка происходит в любом месте строки.
#    Для выполнения скрипта нужно вызвать команду "Run" или "Exec" с именем
#    файла на диске. Например "Run Test.txt". По умолчанию отдаётся 
#    предпочтение запуску из папки пользователя. Для помощи введите "Help".
#  
#


#Изменить цвет фона оболочки на полупрозрачный красный
BGColor 255 0 0 127

#Сменить имя
Name = accuratealx

#Очистить протокол
Clear

#Вывод версии движка
Version

#Установить параметр
SetParam S 'Hello world!'

#Вывод в оболочку строки
Echo @S

#Назначить команду на клавишу Тильда
Attach +Tilde 'Shell On'
