//Описание формата простого синтаксического анализатора

//Управляющие символы
  Divider	= ' '	- Разделитель для частей команды (пробел)
  Staple	= '		- Кавычка "'"
  Control	= `		- Экранирующий символ


//Общая форма записи команды:

  CmdName 'Param one' Param_two Param' 'Three ParamN


//Описание
Алгоритм разбирает строковую команду на массив строк, по следующим правилам:
Разделителем между частями команды является пробел " ". Если в часть команды
необходимо вставить управляющие символы, то такая часть может быть помещена
в одинарные кавычки "'" или экранирована "`". Управляющие символы при чтении
пропускаются.


//Примеры
LoadFromFile 'C:\Read me.txt'
LoadFromFile C:\Read` me.txt
'Concat str' String1 String2 'String N'
