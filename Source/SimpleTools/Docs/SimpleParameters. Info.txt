//Описание формата файла простых настроек


//Управляющие символы
  Commentary = "#"	- Символ заметки
  Divider    = "="	- Разделитель между именем параметра и значением
  Staple     = "'"	- Кавычки "'"
  Control    = '`'	- Экранирующий символ


//Общая форма записи параметра:

  [ParamName]=[Value]


//Описание
Алгоритм определяет имя параметра от начала строки до служебного символа "=". Лишние пробелы
в имени не учитываются. Имя параметра не должно начинатся со служебного символа "#". Любая
строка начинающаяся с "#" считается заметкой и не учитывается. Заметка не должна находится
на одной строке с параметром. Любые символы в кавычках, считаются за обычный символ. Если
в значение параметра необходимо указать один или несколько служебных символов, то такой
параметр может обрамляться кавычками "'" или экранироваться "`". Лишние пробелы в значении
параметра не учитаваются, если только они явно не экранированы "`" или взяты в кавычки "'".
В имени параметра недопустимо использовать управляющие элементы. При сохранении в файл не
происходит экранирование управляющих элементов, однако при чтении учитываются управляющие
символы. Например: 'А=0'=10. При чтении имя параметра будет определено как "А=0".



//Примеры
  #Это заметка
  String1 = accuratealx@gmail.com
  String2 = Это строка с пробелами и без кавычек
  String3 = 'Это строка с символом = и ` в кавычках'
  String4 = Это строка с кавычкой `'
  Hello String = Пробел в имени
  Integer = 123
  Float = 3.14
  B o o l e a n = 0
